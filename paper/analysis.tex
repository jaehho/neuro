\section{Base Neuronal Dynamics} \label{sec:interspike}

The presynaptic input signals decay as
\[
    \frac{dI_s}{dt} = -\frac{I_s}{\tau_s}, \qquad
    \frac{dx_{\mathrm{pre}}}{dt} = -\frac{x_{\mathrm{pre}}}{\tau_+}, \qquad
    \frac{dr_{\mathrm{pre}}}{dt} = -\frac{r_{\mathrm{pre}}}{\tau_r}.
\]
These are fully determined by initial conditions set at the last presynaptic spike and enter the postsynaptic system as forcing functions. The postsynaptic dynamics are:
\begin{align}
    \tau_m \frac{dV}{dt} &= -(V - E_L) + R_m\, w\, I_s(t), \label{eq:inter_V} \\[6pt]
    \frac{dy_{\mathrm{post}}}{dt} &= -\frac{y_{\mathrm{post}}}{\tau_-}, \label{eq:inter_ypost} \\[6pt]
    \frac{dE}{dt} &= -\frac{E}{\tau_e}, \label{eq:inter_E} \\[6pt]
    \frac{dr_{\mathrm{post}}}{dt} &= -\frac{r_{\mathrm{post}}}{\tau_r}, \label{eq:inter_rpost} \\[6pt]
    \tau_{\bar{R}} \frac{d\bar{R}}{dt} &= -\bar{R} - \bigl(r_{\mathrm{post}} - \tfrac{1}{2}\, r_{\mathrm{pre}}\bigr)^2, \label{eq:inter_Rbar} \\[6pt]
    \frac{dw}{dt} &= \Bigl[-\bigl(r_{\mathrm{post}} - \tfrac{1}{2}\, r_{\mathrm{pre}}\bigr)^2 - \bar{R}\Bigr]\, E. \label{eq:inter_w}
\end{align}

The postsynaptic state vector is $\mathbf{x}(t) = \bigl(V,\; y_{\mathrm{post}},\; E,\; r_{\mathrm{post}},\; \bar{R},\; w\bigr) \in \mathbb{R}^6$, and the inputs $\bigl(I_s(t),\; x_{\mathrm{pre}}(t),\; r_{\mathrm{pre}}(t)\bigr)$.

\section{Pre-synaptic spike event} \label{sec:pre-spike}

At each presynaptic spike time $t_{\mathrm{pre}}^k$, the Dirac term $\rho_{\mathrm{pre}}(t) = \delta(t - t_{\mathrm{pre}}^k)$ fires instantaneously. The input subsystem variables receive impulsive increments, which are implemented as discrete jumps in the state:
\[
    I_s \;\leftarrow\; I_s + \frac{1}{\tau_s}, \qquad
    x_{\mathrm{pre}} \;\leftarrow\; x_{\mathrm{pre}} + \frac{1}{\tau_+}, \qquad
    r_{\mathrm{pre}} \;\leftarrow\; r_{\mathrm{pre}} + \frac{1}{\tau_r}.
\]
The only postsynaptic state variable that receives a direct impulsive drive at a presynaptic spike is the eligibility trace, which acquires an LTD contribution through the postsynaptic trace $y_{\mathrm{post}}$ evaluated at $t_{\mathrm{pre}}^k$:
\[
    E \;\leftarrow\; E - A_-(w)\, y_{\mathrm{post}}(t_{\mathrm{pre}}^{k-}).
\]
All other postsynaptic variables ($V$, $y_{\mathrm{post}}$, $r_{\mathrm{post}}$, $\bar{R}$, $w$) are continuous through $t_{\mathrm{pre}}^k$ and follow their inter-spike dynamics \eqref{eq:inter_V}--\eqref{eq:inter_w}, now driven by the updated $I_s$.

\section{Post-synaptic spike event} \label{sec:post-spike}

At each postsynaptic spike time $t_{\mathrm{post}}^k$, defined by the threshold crossing 
\begin{equation} \label{eq:threshold}
    t_{\mathrm{post}}^k : \quad V(t_{\mathrm{post}}^k) = \theta \quad \text{and} \quad \left.\frac{dV}{dt}\right|_{t=t_{\mathrm{post}}^k} > 0,
\end{equation}
the membrane potential is reset and a discrete jump is applied to the postsynaptic state variables $V$, $y_{\mathrm{post}}$, $r_{\mathrm{post}}$ according to the reset map:
\[
    V \;\leftarrow\; V_{\mathrm{reset}}, \qquad
    y_{\mathrm{post}} \;\leftarrow\; y_{\mathrm{post}} + \frac{1}{\tau_-}, \qquad
    r_{\mathrm{post}} \;\leftarrow\; r_{\mathrm{post}} + \frac{1}{\tau_r}.
\]
and the eligibility trace acquires an LTP contribution through the presynaptic trace $x_{\mathrm{pre}}$ evaluated at $t_{\mathrm{post}}^k$:
\[
    E \;\leftarrow\; E + A_+(w)\, x_{\mathrm{pre}}(t_{\mathrm{post}}^{k-}).
\]
The variables $\bar{R}$ and $w$ are continuous through $t_{\mathrm{post}}^k$; they are driven by the smoothly varying rate estimates $r_{\mathrm{post}}$ and $r_{\mathrm{pre}}$. Following the spike, the postsynaptic voltage dynamics are suspended for a refractory period $\tau_{\mathrm{ref}}$, after which integration resumes from $V_{\mathrm{reset}}$.

\section{Complete Dynamics} \label{sec:reduced_system}

\begin{align}
    \tau_m \frac{dV}{dt} &= -(V - E_L) + R_m\, w\, I_s(t), \label{eq:reduced_V} \\[6pt]
    \tau_- \frac{dy_{\mathrm{post}}}{dt} &= -y_{\mathrm{post}} + \rho_{\mathrm{post}}(t), \label{eq:reduced_ypost} \\[6pt]
    \tau_e \frac{dE}{dt} &= -E + \eta_+(w_{\mathrm{max}} - w)\, x_{\mathrm{pre}}\, \rho_{\mathrm{post}}(t) - \eta_-\, w\, y_{\mathrm{post}}\, \rho_{\mathrm{pre}}(t), \label{eq:reduced_E} \\[6pt]
    \tau_r \frac{dr_{\mathrm{post}}}{dt} &= -r_{\mathrm{post}} + \rho_{\mathrm{post}}(t), \label{eq:reduced_rpost} \\[6pt]
    \tau_{\bar{R}} \frac{d\bar{R}}{dt} &= -\bar{R} - \bigl(r_{\mathrm{post}} - \tfrac{1}{2}\, r_{\mathrm{pre}}\bigr)^2, \label{eq:reduced_Rbar} \\[6pt]
    \frac{dw}{dt} &= \Bigl[-\bigl(r_{\mathrm{post}} - \tfrac{1}{2}\, r_{\mathrm{pre}}\bigr)^2 - \bar{R}\Bigr]\, E, \label{eq:reduced_w}
\end{align}
